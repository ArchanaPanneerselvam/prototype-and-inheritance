Question 1:

Create 2 objects - parent and child. Attach methods to parent and use those methods in child object using parents prototype

        class car{
            constructor(wheel,color){
                this.wheel=wheel
                this.color=color
            }
        }
        class jaguar extends car{
            constructor(wheel,color,engine,make,seats){
                super(wheel,color)
                this.engine=engine
                this.make=make
                this.seats=seats
            }
        }
        var newcar=new jaguar(4,"blue","1l","F-Pace",5)
        console.log(newcar);
